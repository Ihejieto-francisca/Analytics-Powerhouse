import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Generate synthetic data
np.random.seed(42)
dates = pd.date_range(start="2023-01-01", end="2023-12-31", freq="ME")  # Use 'ME' instead of 'M'
regions = ['North', 'South', 'East', 'West']
products = ['Electronics', 'Apparel', 'Home', 'Beauty']
data = {
    'Date': np.repeat(dates, 4),
    'Region': np.tile(regions, len(dates)),
    'Product': np.random.choice(products, len(dates)*4),
    'Sales': np.random.randint(100, 1000, len(dates)*4),
    'Customer_Age': np.random.randint(18, 65, len(dates)*4),
    'Marketing_Spend': np.random.uniform(500, 2000, len(dates)*4),
    'Profit': np.random.uniform(-50, 500, len(dates)*4)
}
df = pd.DataFrame(data)
# Example 1: Monthly Sales Trend (Line Chart)
plt.figure(figsize=(10, 5))
df.groupby('Date')['Sales'].sum().plot(marker='o', linestyle='--', color='blue')
plt.title('Monthly Sales Trend (2023)')
plt.xlabel('Month')
plt.ylabel('Total Sales')
plt.grid(True)
plt.savefig("monthly trend.png")

plt.show()  # Use plt.show() to render the plot
